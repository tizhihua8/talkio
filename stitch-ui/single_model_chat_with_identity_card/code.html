Sure! Here's a concise custom hook `useFetch` written in TypeScript that handles loading and error states.

typescript content_copy 
function useFetch<T>(url: string) {
  const [data, setData] = useState<T | null>(null);
  const [loading, setLoading] = useState(true);
  useEffect(() => {
    fetch(url)
      .then(res => res.json())
      .then(d => {
        setData(d);
        setLoading(false);
      });
  }, [url]);
  return { data, loading };
}
